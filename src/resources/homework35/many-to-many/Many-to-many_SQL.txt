--Many to many
CREATE TABLE Actor(
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    firstname varchar,
    lastname varchar,
    age int
);

CREATE TABLE Movie(
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name varchar,
    description varchar
);

CREATE TABLE Actor_Movie(
    actor_id int REFERENCES Actor(id),
    movie_id int REFERENCES Movie(id)
)
